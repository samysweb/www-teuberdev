{{- $entries := site.Data.cv.service.entries | default (slice) -}}

{{- /* Build ordered list of role keys and a map role -> label */}}
{{- $roleOrder := slice "conference_review" "subreview" "journal_review" -}}
{{- $roleLabels := dict
      "conference_review" "Conference Reviews"
      "subreview"         "Subreviews"
      "journal_review"    "Journal Reviews" -}}

{{- /* Collect custom roles in order of first appearance */}}
{{- range $entries -}}
  {{- if and (eq .role "custom") .role_custom -}}
    {{- if not (in $roleOrder .role_custom) -}}
      {{- $roleOrder = $roleOrder | append .role_custom -}}
      {{- $roleLabels = merge $roleLabels (dict .role_custom .role_custom) -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

{{- /* Group entries by role key */}}
{{- $groups := dict -}}
{{- range $entries -}}
  {{- $key := .role -}}
  {{- if eq .role "custom" -}}{{- $key = .role_custom | default "custom" -}}{{- end -}}
  {{- $existing := index $groups $key | default (slice) -}}
  {{- $groups = merge $groups (dict $key ($existing | append .)) -}}
{{- end -}}

\CVSection{Professional Service}
\begin{itemize}[leftmargin=*]
{{- range $roleKey := $roleOrder -}}
  {{- $items := index $groups $roleKey -}}
  {{- if $items -}}
    {{- $label := index $roleLabels $roleKey | default $roleKey -}}
    {{- $parts := slice -}}
    {{- range $items -}}
      {{- $entry := partial "cv/tex_escape.html" .venue -}}
      {{- $entry = printf "%s (%s)" $entry (partial "cv/tex_escape.html" .year) -}}
      {{- with .note -}}{{- $entry = printf "%s, %s" $entry (partial "cv/tex_escape.html" .) -}}{{- end -}}
      {{- $parts = $parts | append $entry -}}
    {{- end }}
\item \textbf{ {{ partial "cv/tex_escape.html" $label }} }: {{ delimit $parts "; " }}
  {{- end -}}
{{- end }}
\end{itemize}
